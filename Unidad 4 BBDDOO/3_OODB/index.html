            System.err.println("Error de ejecuci贸n:");
            e.printStackTrace();
        } finally {
            // 5. Cerrar recursos
            if (em != null && em.isOpen()) {
                em.close();
                System.out.println("EntityManager cerrado");
            }
            if (emf != null && emf.isOpen()) {
                emf.close();
                System.out.println("EntityManagerFactory cerrado");
            }
        }
    }
